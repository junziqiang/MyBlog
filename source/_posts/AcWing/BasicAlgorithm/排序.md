# 快速排序
## 快速排序模板
给定你一个长度为 n 的整数数列。

请你使用快速排序对这个数列按照从小到大进行排序。

并将排好序的数列按顺序输出。

输入格式
输入共两行，第一行包含整数 n。

第二行包含 n 个整数（所有整数均在 1∼109 范围内），表示整个数列。

输出格式
输出共一行，包含 n 个整数，表示排好序的数列。

数据范围
1≤n≤100000
输入样例：
5
3 1 2 4 5
输出样例：
1 2 3 4 5

```cpp
/**
 * @file acwing785.cpp
 * @author your name (you@domain.com)
 * @brief 
 * @version 0.1
 * @date 2022-10-27
 * 
 * @copyright Copyright (c) 2022
 * 快速排序
 */
#include<bits/stdc++.h>
using namespace std;

void quickSort(vector<int>& vec, int left, int right)
{
    if (left >= right) return;
    int l = left - 1, r = right + 1, x = vec[l + r >> 1];
    while (l < r) {
        while (vec[++l] < x);
        while (vec[--r] > x);
        if (l < r) swap(vec[l], vec[r]);
    }
    quickSort(vec, left, r);
    quickSort(vec, r + 1, right);
}
int main()
{
    int n;
    cin >> n;
    vector<int> vec(n);
    for (int i = 0; i < n; i++) {
        cin >> vec[i];
    }
    quickSort(vec, 0, n  - 1);
    for (int i = 0; i < n; i++) {
        cout << vec[i] << " ";
    }
}
```
## 第k个数
给定一个长度为 n 的整数数列，以及一个整数 k，请用快速选择算法求出数列从小到大排序后的第 k 个数。

输入格式
第一行包含两个整数 n 和 k。

第二行包含 n 个整数（所有整数均在 1∼109 范围内），表示整数数列。

输出格式
输出一个整数，表示数列的第 k 小数。

数据范围
1≤n≤100000,
1≤k≤n
输入样例：
5 3
2 4 1 5 3
输出样例：
3
```cpp
#include<bits/stdc++.h>
using namespace std;

int quickSort(vector<int>& vec, int left, int right, int k)
{
    if (right == left) return vec[left];
    int l = left - 1, r = right + 1, x = vec[left + right >> 1];
    while (l < r) {
        while (vec[++l] < x);
        while (vec[--r] > x);
        if (l < r) swap(vec[l], vec[r]);
    }
    int length = r - left + 1;
    if (k <= length) quickSort(vec, left, r, k);
    else quickSort(vec, r + 1, right, k - length);
}

int main()
{
    int n, k;
    cin >> n >> k;
    vector<int> vec(n);;
    for (int i = 0; i < n; i++) cin >> vec[i];
    
    cout << quickSort(vec, 0, n - 1, k) << endl;
}
```

# 归并排序
## 归并排序模板
给定你一个长度为 n 的整数数列。

请你使用归并排序对这个数列按照从小到大进行排序。

并将排好序的数列按顺序输出。

输入格式
输入共两行，第一行包含整数 n。

第二行包含 n 个整数（所有整数均在 1∼109 范围内），表示整个数列。

输出格式
输出共一行，包含 n 个整数，表示排好序的数列。

数据范围
1≤n≤100000
输入样例：
5
3 1 2 4 5
输出样例：
1 2 3 4 5

```cpp
#include<bits/stdc++.h>
using namespace std;

void mergeSort(vector<int>& vec, int left, int right)
{
    if (right <= left) return;
    int mid = left + right >> 1;
    mergeSort(vec, left, mid);
    mergeSort(vec, mid + 1, right);
    int l = left, r = mid + 1;
    vector<int> tmpVec;
    while (l <= mid && r <= right) {
        if (vec[l] < vec[r]) {
            tmpVec.push_back(vec[l++]);
        } else {
            tmpVec.push_back(vec[r++]);
        }
    }
    while (l <= mid) tmpVec.push_back(vec[l++]);
    while (r <= right) tmpVec.push_back(vec[r++]);
    for (int pos = 0; pos < tmpVec.size(); pos++) {
        vec[left + pos] = tmpVec[pos];
    }
}

int main()
{
    int n;
    cin >> n;
    vector<int> vec(n);
    for (int i = 0; i < n; i++) {
        cin >> vec[i];
    }
    mergeSort(vec, 0, vec.size() - 1);
    for (int item : vec) {
        cout << item << " ";
    }
}
```

## 逆序对的数量
给定一个长度为 n 的整数数列，请你计算数列中的逆序对的数量。

逆序对的定义如下：对于数列的第 i 个和第 j 个元素，如果满足 i<j 且 a[i]>a[j]，则其为一个逆序对；否则不是。

输入格式
第一行包含整数 n，表示数列的长度。

第二行包含 n 个整数，表示整个数列。

输出格式
输出一个整数，表示逆序对的个数。

数据范围
1≤n≤100000，
数列中的元素的取值范围 [1,109]。

输入样例：
6
2 3 4 5 6 1
输出样例：
5
```cpp
#include<bits/stdc++.h>
using namespace std;

/**
 * @brief 逆序对的数量
 * 
 */
long long mergeSort(vector<int>& vec, int left, int right)
{
    if (right <= left) {
        return 0;
    }
    int mid = left + right >> 1;
    long long res = mergeSort(vec, left, mid) + mergeSort(vec, mid + 1, right);
    vector<int> tmpVec;
    int l = left;
    int r = mid + 1;
    while (l <= mid && r <= right) {
        if (vec[l] <= vec[r]) {
            tmpVec.push_back(vec[l++]);
        } else {
            tmpVec.push_back(vec[r++]);
            res += mid - l + 1;
        }
    }
    while (l <= mid) tmpVec.push_back(vec[l++]);
    while (r <= right) tmpVec.push_back(vec[r++]);
    for (int pos = 0; pos < tmpVec.size(); pos++) {
        vec[left + pos] = tmpVec[pos];
    }
    return res;
}
int main()
{
    int n;
    cin >> n;
    vector<int> vec(n);
    for (int i = 0; i < n; i++) {
        cin >> vec[i];
    }
    cout << mergeSort(vec, 0, vec.size() - 1) << endl;
}
```
